```html
<template>
    <div class="app-container">
         
        <div>
             
            <h1>这是App组件</h1>
            <StudyDemo></StudyDemo>
        </div>
        <div><StudyDemo :title="msg"></StudyDemo> </div>
    </div>
</template>

<script>
    import StudyDemo from './components/Study05/StudyDemo.vue'

    export default {
        components: {
            StudyDemo,
        },

        data() {
            return {
                msg: '这是App组件中的msg',
            }
        },
    }
</script>
<style></style>

//子组件
<template>
     
    <div>
           
        <h1>这是Demo01组件</h1>
           
        <p v-if="title">{{ title }}</p>
         
    </div>
</template>
<script>
    export default {
        props: {
            title: {
                type: String,
                default: '',
            },
        },
    }
</script>
```
