# EventBus.js

```js
import Vue from 'vue'
const Bus = new Vue()
export default Bus
```

# StudyDemo01.vue

```html
<template>
    <div>
        <demo02></demo02>
        <demo03></demo03>
    </div>
</template>

<script>
    import demo02 from './StudyDemo02.vue'
    import demo03 from './StudyDemo03.vue'
    export default {
        components: {
            demo02,
            demo03,
        },
    }
</script>
```

# StudyDemo02.vue

```html
<template>
     
    <div>
         
        <h1><button @click="sendMsg">发送消息</button> </h1>
         
    </div>
</template>

<script>
    import Bus from '../../utils/EventBus'
    export default {
        methods: {
            sendMsg() {
                Bus.$emit('sendMsg', 'hello vue')
            },
        },
    }
</script>
```

# StudyDemo03.vue

```html
<template>
    <div>
           
        <h1>demo03</h1>
         
    </div>
</template>

<script>
    import Bus from '../../utils/EventBus'
    export default {
        created() {
            Bus.$on('sendMsg', (msg) => {
                console.log(msg)
            })
        },
    }
</script>
```
