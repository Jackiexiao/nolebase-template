```html
<template>
     
    <div>
           
        <h1 v-color="color">颜色</h1>
         
    </div>
</template>

<script>
    export default {
        directives: {
            color: {
                //元素被添加到页面时触发
                inserted(el, binding) {
                    el.style.color = binding.value
                },
                //指令的值修改时触发
                update(el, binding) {
                    el.style.color = binding.value
                },
            },
        },
        data() {
            return {
                color: 'red',
            }
        },
    }
</script>
```
