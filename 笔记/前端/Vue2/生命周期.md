# vue生命周期函数（钩子函数）

![](../../img/Pasted%20image%2020251118155637.png)
![](../../img/Pasted%20image%2020251118155803.png)

```html
<script>
    const app = new Vue({
        el: '#app',
        data: {
            count: 105,
        },
        //创建阶段（准备数据）
        beforeCreate() {
            console.log('beforeCreate 响应式数据准备之前' + this.count)
        },
        created() {
            console.log('created 响应式数据准备之后' + this.count)
            //可以开始发送初始化渲染请求了
        },

        //挂载阶段（渲染模板）
        beforeMount() {
            console.log(
                'beforeMount 渲染模板之前',
                document.querySelector('span').innerHTML
            )
        },
        mounted() {
            console.log(
                'mounted 渲染模板之后',
                document.querySelector('span').innerHTML
            )
            //可以操作dom元素了
        },

        //更新阶段（修改数据->更新视图）
        beforeUpdate() {
            console.log(
                'beforeUpdate 更新之前',
                document.querySelector('span').innerHTML
            )
        },
        updated() {
            console.log(
                'updated 更新之后',
                document.querySelector('span').innerHTML
            )
        },

        //销毁阶段
        beforeDestroy() {
            console.log('beforeDestroy 销毁之前')
            //清除掉vue以外的资源占用，定时器，延时器
        },
        destroyed() {
            console.log('destroyed 销毁之后')
        },
    })
</script>
```
