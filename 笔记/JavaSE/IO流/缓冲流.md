## 字节缓冲流

![Pasted image 20251107111152](../../img/Pasted%20image%2020251107111152.png)
![Pasted image 20251107111203](../../img/Pasted%20image%2020251107111203.png)

```Java
//一次读取一个字节
BufferedInputStream bis = new BufferedInputStream(new
            FileInputStream("F:\\JavaDevelopment\\JavaSE\\src\\javaStudy\\MyIO\\a.txt"));
    BufferedOutputStream bos = new BufferedOutputStream(new
            FileOutputStream("F:\\JavaDevelopment\\JavaSE\\src\\javaStudy\\MyIO\\b.txt"));
    int len;
    while ((len = bis.read()) != -1) {
        bos.write(len);
    }
    bos.close();
    bis.close();
}

//一次读取多个字节
BufferedInputStream bis = new BufferedInputStream(
        new FileInputStream("src/javaStudy/MyIO/a.txt"));
BufferedOutputStream bos = new BufferedOutputStream(
        new FileOutputStream("src/javaStudy/MyIO/c.txt"));
byte[] bytes = new byte[1024];
int len;
while ((len = bis.read(bytes)) != -1) {
    bos.write(bytes, 0, len);
}
bos.close();
bis.close();
```
