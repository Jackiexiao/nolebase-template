字符流底层有缓冲区，想要自动刷新需要开启
![](../../img/Pasted%20image%2020251113133120.png)
![](../../img/Pasted%20image%2020251113133131.png)

```Java
public class IODemo31 {
    public static void main(String[] args) throws FileNotFoundException {
        PrintWriter pw = new PrintWriter(
                new FileOutputStream("src/javaStudy/MyIO/d.txt"),
                true, Charset.forName("UTF-8"));
                pw.println("张三");
                pw.print("你好，中国");
                pw.printf("你好，%s",true);
                // 写入ASCII码97对应的字符a
                pw.write(97);
                pw.close();
    }
}
```

```Java
public class IODemo32 {
    public static void main(String[] args) {
        //获取打印流的对象，此打印流在虚拟机启动时，由虚拟机创建，默认指向控制台
        //特殊的打印流，系统中的标准输出流，是不能关闭的，在系统中是唯一的。
        PrintStream ps = System.out;
        //调用打印流中的println方法
        //写出数据，自动刷新，自动换行
        ps.println("你好，中国");
    }
}
```
