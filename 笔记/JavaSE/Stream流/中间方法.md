ArrayList<String> list = new ArrayList<>();
Collections.addAll(list,"张三-18","李四-19","王五-20");
list.stream().map( s -> {
        String ageStr = s.split("-")[1];
        int age = Integer.parseInt(ageStr);
        return age;
    }
).forEach(s -> System.out.println(s));
