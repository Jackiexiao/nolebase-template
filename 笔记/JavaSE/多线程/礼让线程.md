public class MyThread06 extends Thread{
    @Override
    public void run() {
        for (int i = 0; i < 100; i++) {
            System.out.println(getName() + "运行，i=" + i);
            Thread.yield();
        }
    }
}

public class MultiDemo07 {
    public static void main(String[] args) {
        MyThread06 t1 = new MyThread06();
        MyThread06 t2 = new MyThread06();
        t1.setName("线程1");
        t2.setName("线程2");
        t1.start();
        t2.start();
    }
}
