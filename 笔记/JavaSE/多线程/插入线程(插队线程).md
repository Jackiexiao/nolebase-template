public class MultiDemo08 {
    public static void main(String[] args) throws InterruptedException {
        MyThread04 t = new MyThread04();
        t.setName("线程1");
        t.start();
        //表示把这个线程，插入到当前线程之前。
        //t：线程1
        //当前线程：main线程
        t.join();

        for (int i = 0; i < 18; i++) {
            System.out.println("main线程运行，i=" + i);
        }
    }
}
