	String str2 = "Java自从95年问世以来，经历了很多版本，目前企业中用的最多的是ava8和Java11，因为这\n" +

	Pattern p = Pattern.compile("Java\\d{0,2}");  //获取一个正则表达式

	while (m.find()){  //find找到返回true，否则返回false
	    System.out.println(m.group());  //group会将截取的字符串返回
	}
