public class MyReflect04 {
    public static void main(String[] args) throws ClassNotFoundException {
        Class<?> aClass = Class.forName("javaStudy.myreflect.Student");
        //获取所有的成员方法
        Method[] declaredMethods = aClass.getDeclaredMethods();
        for (Method declaredMethod : declaredMethods) {
            System.out.println(declaredMethod);
        }

        System.out.println("-----------------");
		//获取方法抛出的异常
		Method getName = aClass.getMethod("getName", null);
		Class<?>[] exceptionTypes = getName.getExceptionTypes();
		System.out.println(getName);
		for (Class<?> exceptionType : exceptionTypes) {
		    System.out.println(exceptionType);
		}

		//通过反射运行方法
		Method printInfo = aClass.getDeclaredMethod("printInfo", String.class);
		printInfo.setAccessible(true);
		Student s = new Student();
		printInfo.invoke(s, "张三");
    }
}
